    \def\formule@start#1{%
      \formule@startline
      \if\noexpand#1\noexpand~%
        \let\next\formule@
      \else \def\next{\formule@#1}\fi
      \next}%
    \def\formule@#1~{\formule@@#1!end\@nil}%
    \def\formule@@#1!end{%
       \formule@addtoline{#1}%
       \futurelet\next\formule@@@}%
    \def\formule@@@#1\@nil{%
       \ifx\next\@nil
         \formule@processline
         \formule@startline
         \let\next\formule@
       \else
         \def\@tempa##1!end\@nil{##1}%
         \@temptokena{!end}%
         \def\next{\expandafter\formule@test\@tempa#1\@nil~}%
       \fi \next}%
    \def\formule@test#1{%
           \let\next\formule@test
           \if\noexpand#1\noexpand~%
             \expandafter\formule@addtoline
               \expandafter{\the\@temptokena}%
             \formule@processline
             \formule@startline
             \let\next\formule@
           \else \if\noexpand#1
             \@temptokena\expandafter{\the\@temptokena#1}%
           \else \if\noexpand#1\noexpand[%
             \let\@tempc\@empty
             \let\next\formule@testend
           \else
             \expandafter\formule@addtoline
               \expandafter{\the\@temptokena}%
             \def\next{\formule@#1}%
           \fi\fi\fi
           \next}%
    \def\formule@testend#1{%
         \if\noexpand#1\noexpand~%
           \expandafter\formule@addtoline
             \expandafter{\the\@temptokena[}%
           \expandafter\formule@addtoline
             \expandafter{\@tempc}%
           \formule@processline
           \formule@startline
           \let\next\formule@
         \else\if\noexpand#1\noexpand]%
           \let\next\formule@@testend
         \else\if\noexpand#1\noexpand!%
           \expandafter\formule@addtoline
             \expandafter{\the\@temptokena[}%
           \expandafter\formule@addtoline
             \expandafter{\@tempc}%
           \def\next{\formule@!}%
         \else \expandafter\def\expandafter\@tempc\expandafter
           {\@tempc#1}\fi\fi\fi
         \next}%
    \def\formule@@testend{%
       \ifx\@tempc\@currenvir
         \formule@finish
         \edef\next{\noexpand\end{\@currenvir}}%
       \else
         \expandafter\formule@addtoline
           \expandafter{\the\@temptokena[}%
           \expandafter\formule@addtoline
             \expandafter{\@tempc]}%
         \let\next\formule@
       \fi
       \next}}
